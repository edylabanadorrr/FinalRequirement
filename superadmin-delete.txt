<?php
header("Access-Control-Allow-Origin: *");
header("Access-Control-Allow-Methods: DELETE");
header("Access-Control-Allow-Headers: Origin, X-Requested-With, Content-Type, Accept");

$host = 'localhost';
$user_name = 'root';
$pass_word = '';
$database = 'ionic-batelec';

$conn = mysqli_connect($host, $user_name, $pass_word, $database);

if ($_SERVER['REQUEST_METHOD'] === 'DELETE') {
    $consumerID = $_GET['ConsumerID'];

    if (empty($consumerID)) {
        $response = array('status' => 'error', 'message' => 'Invalid consumer ID');
        echo json_encode($response);
        exit;
    }

    $consumerID = mysqli_real_escape_string($conn, $consumerID);

    $query = "DELETE FROM consumertable WHERE ConsumerID = '$consumerID'";

    if (mysqli_query($conn, $query)) {
        $response = array('status' => 'success', 'message' => 'Customer deleted successfully');
        echo json_encode($response);
    } else {
        $response = array('status' => 'error', 'message' => 'Failed to delete customer', 'error' => mysqli_error($conn));
        echo json_encode($response);
    }
} else {
    $response = array('status' => 'error', 'message' => 'Invalid request method');
    echo json_encode($response);
}
?>
